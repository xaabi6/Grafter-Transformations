031      (graph-fn [{:keys [Name Description Province Town Address CoordenatesX CoordenatesY Latitude Longitude quality-uri] :as row}]
032          (graph (base-graph "Air Quality Basque Country 2017")

035                  ;;[qb:attribute (langSp Name)]

039                  [qb:measureType CoordenatesX]
040                  [qb:measureType CoordenatesY]
041                  ["http://www.w3.org/2003/01/geo/wgs84_pos#lat" Latitude]
042                  ["http://www.w3.org/2003/01/geo/wgs84_pos#long" Longitude]
043              ]
044          )

049  (def make-celica-graph
050      (graph-fn [{:keys [brand name generation startedProduction finishedProduction
051                         platform engineCode version engineSize valves maxTorque traction length width height curbWeight fuelType fuelTank
052                         numberAirbags numberDoors price celica-uri data-uri] :as row}]
053                (graph (base-graph "Celica")
054                      [celica-uri
055                          [rdf:a "http://dbpedia.org/resource/Car"]
056                          [rdfs:comment (commentary (str "This info is about the " brand " " name " " generation " gen, which engine version is " version))]
057                          [car-manufacturer (s brand)]
058                          [car-name (s name)]
059                          [start-manufacture (integer startedProduction)]
060                          [end-manufacture (integer finishedProduction)]
061                          [platform-types (s platform)]
062                          [engine-code (s engineCode)]
063                          [car-version (s version)]
064                          [engine-size (s (add-cubic-centimetre engineSize))]
065                          [engine-valves (s valves)]
066                          [engine-maxTorque (s (add-revolutions-per-minute maxTorque))]
067                          [car-traction (s traction)]
068                          [measurements-length (s (add-millimeters length))]
069                          [measurements-width (s (add-millimeters width))]
070                          [measurements-height (s (add-millimeters height))]
071                          [measurements-curbWeight (s (add-kilograms curbWeight))]
072                          [car-fuel fuelType]
073                          [car-fuel-tank (s (add-litres fuelTank))]
074                          [car-airbags (integer numberAirbags)]
075                          [car-doors (integer numberDoors)]
076                          [car-price (s (add-euro-symbol price))]
077                      ]
078                      [data-uri
079                          [rdfs:comment (commentary (str "All the data references are taken from one model, even though, there are more") )]
080                      ]
081                )
082      )

119  ;; AIR QUALITY
120  (defn convert-air-quality-to-data
121    "Pipeline to convert tabular people data into a different tabular format."
122    [data-file]
123    (-> (read-dataset data-file)
124        (make-dataset move-first-row-to-header)
125        (make-dataset [:Name :Description :Province :Town :Address :CoordenatesX :CoordenatesY :Latitude :Longitude])
126        (mapc {
127                 :Latitude  parseValue
128                 :Longitude parseValue
129                 :CoordenatesX  parseValue
130                 :CoordenatesY parseValue
131                 :Address removeSymbols
132              }
133        )
134        (derive-column :quality-uri [:Name] quality-uri)
135    )
136  )
