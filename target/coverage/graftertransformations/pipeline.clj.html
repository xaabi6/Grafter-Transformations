166  (defn convert-air-quality-to-data
167    "Pipeline to convert tabular air quality data into a different tabular format."
168    [data-file]
169    (-> (read-dataset data-file)
170        (make-dataset move-first-row-to-header)
171        (make-dataset [:Name :Description :Province :Town :Address :CoordenatesX :CoordenatesY :Latitude :Longitude])
172        (mapc {
173                 :Latitude  parseValue
174                 :Longitude parseValue
175                 :CoordenatesX  parseValue
176                 :CoordenatesY parseValue
177                 :Address removeSymbols
178              }
179        )
180        (derive-column :quality-uri [:Name] quality-uri)
181    )
182  )
