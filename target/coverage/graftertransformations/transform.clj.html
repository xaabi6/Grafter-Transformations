049          (replace "(" "-")
050          (replace ")" "")
051          (replace "  " "")
052          (replace "," "-")
053          (replace "." "")
054          (replace " " "-")
055          (replace "/" "-")
056          (replace "'" "")
057          (replace "---" "-")

059       )

063  ;Convertidor a Float o Integer
064  (defmulti parseValue class)
065  (defmethod parseValue :default            [x] x)
066  (defmethod parseValue nil                 [x] nil)
067  (defmethod parseValue java.lang.Character [x] (Character/getNumericValue x))
068  (defmethod parseValue java.lang.String    [x] (if (= "" x)
069                                                  nil
070                                                  (if (.contains x ".")
071                                                    (Double/parseDouble x)
072                                                    (Integer/parseInt x)
073                                                  )
074                                                ))

078    (when (seq date)
079      (let [[d m y] (st/split date #"/")]
080        (apply str (interpose "-" [y m d] ))
